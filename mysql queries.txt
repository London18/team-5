CREATE TABLE SESSIONS(
SESSIONID INTEGER AUTO_INCREMENT NOT NULL,
PATIENTID VARCHAR(12) NOT NULL,
STARTTD DATETIME NOT NULL,
ENDTD DATETIME NOT NULL,
PRIMARY KEY (SESSIONID));

CREATE TABLE SESSIONSASS(
SESSIONID INTEGER,
PAYROLLNUM VARCHAR(12),
LEFTHOME BOOLEAN,
COMMENT TEXT,
HOME BOOLEAN,
PRIMARY KEY (SESSIONID, PAYROLLNUM),
FOREIGN KEY (SESSIONID) REFERENCES SESSIONS(SESSIONID));

CREATE TABLE STAFF(
PAYROLLNUM VARCHAR(12) NOT NULL,
NAME VARCHAR(250),
ROLE VARCHAR(250),
EMAIL_ADDRESS VARCHAR(250),
PRIMARY KEY (PAYROLLNUM));

CREATE TABLE PASSWORDS(
PAYROLLNUM VARCHAR(12),
USERNAME VARCHAR(25),
HASHPASSWORD VARCHAR(256),
SALT VARCHAR(256),
PRIMARY KEY (PAYROLLNUM));
